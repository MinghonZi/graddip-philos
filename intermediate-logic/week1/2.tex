\begin{itemize}
\item $p \leqslant \{p \land r\}$
\item $q \leqslant \{p \to q, p \land r\}$ and $r \leqslant \{p \land r\}$
\item $q \land r \leqslant \{ p \to q, p \land r \ \text{left}, p \land r \ \text{right} \}$
\item $(p \land r) \to (q \land r) \leq \{ p \to q \}$  
\end{itemize}
